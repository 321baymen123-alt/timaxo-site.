    <script>
        let historyStack = [];

        function switchPanels(hideId, showId) {
            document.getElementById(hideId).classList.add("hidden-panel");
            historyStack.push(hideId);
            document.getElementById(showId).classList.remove("hidden-panel");
            if (showId !== 'start-section') {
                document.getElementById('back-button').classList.remove("hidden-panel");
            }
        }

        function goBack() {
            if (historyStack.length > 0) {
                document.getElementById('main-title').style.display = 'block';
                const currentPanel = document.querySelector('.menu-wrapper:not(.hidden-panel)');
                if (currentPanel) {
                    currentPanel.classList.add("hidden-panel");
                }
                const previousPanelId = historyStack.pop();
                document.getElementById(previousPanelId).classList.remove("hidden-panel");
                if (previousPanelId === 'start-section') {
                    document.getElementById('back-button').classList.add("hidden-panel");
                }
            }
        }

        function showContactSummary() {
            document.querySelectorAll('.menu-wrapper').forEach(p => p.classList.add("hidden-panel"));
            document.getElementById('back-button').classList.add("hidden-panel");
            document.getElementById('main-title').style.display = 'none';
            document.getElementById('contact-section').classList.remove("hidden-panel");
        }
    </script>
